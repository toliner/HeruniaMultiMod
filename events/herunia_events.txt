country_event = {
    id = herunia.1
    title = herunia.1.t
    desc = herunia.1.d
    is_triggered_only = yes
    
    option = {
        name = herunia.1.a
        
        if = {
            limit = {
                amount_research_slots < 4
            }
            set_research_slots = 4
        }
        else = {
            add_research_slot = 1
        }
        if = {
            limit = {
                industrial_complex < 2
            }
            add_offsite_building = {
                type = industrial_complex
                level = 3
            }
        }
        else_if = {
            limit = {
                industrial_complex < 4
            }
            add_offsite_building = {
                type = industrial_complex
                level = 2
            }
        }
        else_if = {
            limit = {
                industrial_complex < 6
            }
            add_offsite_building = {
                type = industrial_complex
                level = 1
            }
        }
        add_ideas = HERUNIA_player_idea
        if = {
            limit = {
                NOT = {
                    tag = ENG
                }
                is_usa_country = no
            }
            inherit_technology = ENG
        }
        if = {
            limit = {
                is_usa_country = yes
            }
            inherit_technology = USA
        }
        add_to_tech_sharing_group = commonwealth_research
        country_event = {
            id = story.100
        }
        country_event = {
            id = herunia.2
        }
    }
}
